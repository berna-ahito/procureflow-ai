  <action>
    Create app/services/approval_engine.py:

    ALLOWED_TRANSITIONS dict — define ALL valid transitions explicitly:
    {
      "draft": ["pending_review"],
      "pending_review": ["pending_approval", "needs_rule"],
      "pending_approval": ["approved", "rejected", "needs_more_info"],
      "needs_rule": ["pending_approval"],
      "needs_more_info": ["draft"],
      "approved": [],  # terminal
      "rejected": [],  # terminal
    }

    validate_transition(current_status: str, new_status: str) -> None:
      If new_status not in ALLOWED_TRANSITIONS.get(current_status, []):
        raise ValueError(f"Invalid transition: {current_status} → {new_status}")

    evaluate_rules(rules: list[ApprovalRule], estimated_cost: float, category: str) -> Optional[ApprovalRule]:
      Filter to is_active=True rules only.
      Sort by priority ASC (lower int = higher priority).
      For each rule:
        if estimated_cost < rule.min_amount: skip
        if rule.max_amount is not None and estimated_cost > rule.max_amount: skip
        if rule.category is not None and rule.category.lower() != category.lower(): skip
        return rule  (first match wins)
      return None

    route_request(db: Session, request: PurchaseRequest) -> PurchaseRequest:
      Load all active ApprovalRules from DB ordered by priority.
      Call evaluate_rules(rules, request.estimated_cost, request.category).
      If match: set request.assigned_role = match.required_role, set request.status = "pending_approval"
      If no match: set request.status = "needs_rule", set request.assigned_role = None
      db.commit(); db.refresh(request)
      Return updated request.

    apply_decision(db: Session, request: PurchaseRequest, decision: str, note: Optional[str]) -> PurchaseRequest:
      Determine new status from decision:
        "approved" → "approved"
        "rejected" → "rejected"
        "needs_more_info" → "needs_more_info"
      Call validate_transition(request.status, new_status) → raises ValueError on invalid
      request.status = new_status
      db.commit(); db.refresh(request)
      Return request.
      (AuditLog written by caller — not here)
  </action>
